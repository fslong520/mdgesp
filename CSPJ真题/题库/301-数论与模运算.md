### 第22题（2020）阅读程序：进制转换（判断题）
```cpp
#include <iostream>
using namespace std;

long long n, ans;
int k, len;
long long d[1000000];

int main() {
    cin >> n >> k;
    d[0] = 0;
    len = 1;
    ans = 0;
    for (long long i = 0; i < n; ++i) {
        ++d[0];
        for (int j = 0; j + 1 < len; ++j) {
            if (d[j] == k) {
                d[j] = 0;
                d[j + 1] += 1;
                ++ans;
            }
        }
        if (d[len - 1] == k) {
            d[len - 1] = 0;
            d[len] = 1;
            ++len;
            ++ans;
        }
    }
    cout << ans << endl;
    return 0;
}
```
- **题型**：程序阅读（判断题）
- **知识点**：数论、进制转换
- **第1题**：若 k=1，则输出 ans 时，len=n。（）
  - A. 正确
  - B. 错误
  - **答案：×**
  - **解析**：k=1时，每次进位len都会增加，最终len=n。

### 第28题（2021）阅读程序：数论函数（判断题）
```cpp
#include <iostream>
using namespace std;

const int n = 100000;
const int N = n + 1;

int m;
int a[N], b[N], c[N], d[N];
int f[N], g[N];

void init() {
    f[1] = g[1] = 1;
    for (int i = 2; i <= n; i++) {
        if (!a[i]) {
            b[m++] = i;
            c[i] = 1, f[i] = 2;
            d[i] = 1, g[i] = i + 1;
        }
        for (int j = 0; j < m && b[j] * i <= n; j++) {
            int k = b[j];
            a[i * k] = 1;
            if (i % k == 0) {
                c[i * k] = c[i] + 1;
                f[i * k] = f[i] / c[i * k] * (c[i * k] + 1);
                d[i * k] = d[i];
                g[i * k] = g[i] * k + d[i];
                break;
            }
            else {
                c[i * k] = 1;
                f[i * k] = 2 * f[i];
                d[i * k] = g[i];
                g[i * k] = g[i] * (k + 1);
            }
        }
    }
}

int main() {
    init();
    int x;
    cin >> x;
    cout << f[x] << ' ' << g[x] << endl;
    return 0;
}
```
- **题型**：程序阅读（判断题）
- **知识点**：数论、线性筛
- **第28题**：若输入不为"1"，把第 13 行删去不会影响输出的结果。（）
  - A. 正确
  - B. 错误
  - **答案：A**
  - **解析**：第13行f[1]=g[1]=1，输入不为1时不影响结果。

- A. 27
- B. 29
- C. 30
- D. 31

### 第13题（2018）10000以内与10000互质的正整数个数？
- A. 2000
- B. 4000
- C. 6000
- D. 8000
- **答案：B**
- **解析**：欧拉函数φ(10000)=10000×(1-1/2)×(1-1/5)=10000×0.5×0.8=4000。
