19. 【油田探测】cpp (100分)
参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

int n, m, area;
char a[105][105];
int vis[105][105];
int dx[] = {-1, -1, -1, 0, 0, 1, 1, 1};
int dy[] = {-1, 0, 1, -1, 1, -1, 0, 1};

void dfs(int y, int x) {
    if (y < 0 || y >= n || x < 0 || x >= m) return;
    if (vis[y][x] || a[y][x] != 'O') return;
    vis[y][x] = 1;
    area++;
    for (int i = 0; i < 8; i++)
        dfs(y + dy[i], x + dx[i]);
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> a[i][j];
    int cnt = 0, mx = 0;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            if (a[i][j] == 'O' && !vis[i][j]) {
                area = 0;
                dfs(i, j);
                cnt++;
                if (area > mx) mx = area;
            }
    if (cnt == 0) cout << "0 0";
    else cout << cnt << " " << mx;
    return 0;
}
```

20. 【岛屿周长】cpp (100分)
参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

int n, m;
int a[105][105];
int vis[105][105];
int dx[] = {-1, 1, 0, 0};
int dy[] = {0, 0, -1, 1};

void dfs(int y, int x, int& perim) {
    if (y < 0 || y >= n || x < 0 || x >= m) return;
    if (vis[y][x] || a[y][x] == 0) return;
    vis[y][x] = 1;
    int cnt = 4;
    for (int i = 0; i < 4; i++) {
        int ny = y + dy[i], nx = x + dx[i];
        if (ny >= 0 && ny < n && nx >= 0 && nx < m && a[ny][nx] == 1)
            cnt--;
    }
    perim += cnt;
    for (int i = 0; i < 4; i++)
        dfs(y + dy[i], x + dx[i], perim);
}

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            cin >> a[i][j];
    int total = 0;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < m; j++)
            if (a[i][j] == 1 && !vis[i][j]) {
                int p = 0;
                dfs(i, j, p);
                total += p;
            }
    cout << total;
    return 0;
}
```
