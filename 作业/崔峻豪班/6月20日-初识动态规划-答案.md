参考代码（第20题 - 采药问题/01背包）：
```cpp
#include <bits/stdc++.h>
using namespace std;
const int MAXN = 105;
const int MAXT = 1005;
int t[MAXN], v[MAXN];
int f[MAXN][MAXT];

int main() {
    int T, n; cin >> T >> n;
    for (int i = 1; i <= n; i++)
        cin >> t[i] >> v[i];
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= T; j++) {
            if (j < t[i])
                f[i][j] = f[i-1][j];
            else
                f[i][j] = max(f[i-1][j], f[i-1][j-t[i]] + v[i]);
        }
    }
    cout << f[n][T] << endl;
    return 0;
}
```
一维优化版：
```cpp
#include <bits/stdc++.h>
using namespace std;
const int MAXT = 1005;
int f[MAXT];

int main() {
    int T, n; cin >> T >> n;
    for (int i = 1; i <= n; i++) {
        int t, v; cin >> t >> v;
        for (int j = T; j >= t; j--)
            f[j] = max(f[j], f[j-t] + v);
    }
    cout << f[T] << endl;
    return 0;
}
```
