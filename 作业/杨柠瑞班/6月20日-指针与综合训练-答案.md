#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    int a[105];
    for (int i = 0; i < n; i++) cin >> a[i];
    int *p = a;
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++) {
            if (*(p + j) < *(p + minIdx)) minIdx = j;
        }
        if (minIdx != i) {
            int temp = *(p + i);
            *(p + i) = *(p + minIdx);
            *(p + minIdx) = temp;
        }
    }
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << *(p + i);
    }
    cout << endl;
    return 0;
}
