#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, g;
    cin >> n >> g;
    int a[105];
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = 0; i < n; i += g) {
        double sum = 0;
        int cnt = 0;
        for (int j = i; j < n && j < i + g; j++) {
            sum += a[j];
            cnt++;
        }
        cout << fixed << setprecision(1) << sum / cnt << endl;
    }
    return 0;
}
```

#include <bits/stdc++.h>

int main() {
    int n;
    cin >> n;
    int male = 0, female = 0, transfer = 0;
    for (int i = 0; i < n; i++) {
        string name;
        char gender;
        cin >> name >> gender;
        if (gender == 'M') male++;
        else female++;
        if (name.size() >= 2 && name.substr(name.size() - 2) == "_T")
            transfer++;
    }
    cout << male << endl << female << endl << transfer << endl;
    return 0;
}
```

    int cnt[101] = {0};
    for (int i = 0; i < m; i++) {
        int x;
        cin >> x;
        cnt[x]++;
    }
    int maxVote = 0, winner = 0;
    for (int i = 1; i <= 100; i++) {
        if (cnt[i] > maxVote) {
            maxVote = cnt[i];
            winner = i;
        }
    }
    cout << winner << endl;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, target;
    cin >> n >> target;
    int a[1005];
    for (int i = 0; i < n; i++) cin >> a[i];
    int *low = a;
    int *high = a + n - 1;
    while (low <= high) {
        int *mid = low + (high - low) / 2;
        if (*mid == target) {
            cout << (mid - a) << endl;
            return 0;
        } else if (*mid < target) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    cout << -1 << endl;
    return 0;
}
