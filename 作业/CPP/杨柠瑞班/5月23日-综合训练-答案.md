17. 【蜗牛爬井】cpp (100分)
参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int h, a, b, day = 0, cur = 0;
    cin >> h >> a >> b;
    while (cur < h) {
        day++;
        cur += a;
        if (cur >= h) break;
        cur -= b;
    }
    cout << day;
    return 0;
}
```
18. 【三角形判断】cpp (100分)
参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int a, b, c;
    cin >> a >> b >> c;
    if (a + b <= c || a + c <= b || b + c <= a) {
        cout << "NO";
    } else if (a == b && b == c) {
        cout << "EQUILATERAL";
    } else if (a == b || a == c || b == c) {
        cout << "ISOSCELES";
    } else {
        cout << "SCALENE";
    }
    return 0;
}
```
19. 【不孪生素数】cpp (100分)
参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

bool isPrime(int x) {
    if (x < 2) return false;
    for (int i = 2; i * i <= x; i++)
        if (x % i == 0) return false;
    return true;
}

int main() {
    int n, cnt = 0;
    cin >> n;
    for (int i = 2; i <= n; i++)
        if (isPrime(i) && !isPrime(i - 2) && !isPrime(i + 2))
            cnt++;
    cout << cnt;
    return 0;
}
```
20. 【相邻相同元素】cpp (100分)
参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

int n, a[505];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];
    int cnt = 0;
    for (int i = 0; i < n - 1; i++)
        if (a[i] == a[i + 1]) cnt++;
    cout << cnt;
    return 0;
}
```
