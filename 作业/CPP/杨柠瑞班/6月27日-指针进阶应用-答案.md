#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    int **a = new int*[n];
    for (int i = 0; i < n; i++) {
        a[i] = new int[m];
        for (int j = 0; j < m; j++) {
            cin >> *(a[i] + j);
        }
    }
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (j > 0) cout << ' ';
            cout << *(a[i] + j);
        }
        cout << endl;
    }
    for (int i = 0; i < n; i++) delete[] a[i];
    delete[] a;
    return 0;
}

17. cpp (100分)

参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

int main() {
    int n, p1, p2;
    cin >> n >> p1 >> p2;
    int a[105];
    for (int i = 0; i < n; i++) cin >> a[i];
    swap(&a[p1], &a[p2]);
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << a[i];
    }
    cout << endl;
    return 0;
}
```
