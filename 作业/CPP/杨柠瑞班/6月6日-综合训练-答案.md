#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    int a[105];
    for (int i = 0; i < n; i++) cin >> a[i];
    int *p = a;
    for (int i = 0; i < n - 1; i++) {
        int minIdx = i;
        for (int j = i + 1; j < n; j++) {
            if (*(p + j) < *(p + minIdx)) minIdx = j;
        }
        if (minIdx != i) {
            int temp = *(p + i);
            *(p + i) = *(p + minIdx);
            *(p + minIdx) = temp;
        }
    }
    cout << *(p + n / 2) << endl;
    return 0;
}

参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;

void count(int *arr, int n, int *pos, int *neg, int *zero) {
    *pos = *neg = *zero = 0;
    for (int i = 0; i < n; i++) {
        if (*(arr + i) > 0) (*pos)++;
        else if (*(arr + i) < 0) (*neg)++;
        else (*zero)++;
    }
}

int main() {
    int n;
    cin >> n;
    int a[105];
    for (int i = 0; i < n; i++) cin >> a[i];
    int pos, neg, zero;
    count(a, n, &pos, &neg, &zero);
    cout << pos << endl << neg << endl << zero << endl;
    return 0;
}
```
