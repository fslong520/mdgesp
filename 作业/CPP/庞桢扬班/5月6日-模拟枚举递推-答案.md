### 第 16 题（编程第1题）：猴子摘桃

```cpp
#include <iostream>
using namespace std;

int main() {
    int M, K;
    cin >> M >> K;

    long long peach = K;
    for (int day = M; day >= 1; day--) {
        peach = (peach + 1) * 2;
    }

    cout << peach << endl;
    return 0;
}
```

### 第 17 题（编程第2题）：发作业本

```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N, M;
    cin >> N >> M;

    vector<int> cnt(N + 1, 0);
    int remaining = M;
    bool stop = false;

    while (!stop && remaining > 0) {
        for (int i = 1; i <= N; i++) {
            if (remaining >= i) {
                cnt[i] += i;
                remaining -= i;
            } else {
                stop = true;    // 不够领，整个分发停止
                break;
            }
        }
    }

    for (int i = 1; i <= N; i++) {
        cout << cnt[i] << " \n"[i == N];
    }
    return 0;
}
```

### 第 18 题（编程第3题）：立方和之谜

```cpp
#include <iostream>
#include <set>
using namespace std;

int main() {
    int N;
    cin >> N;

    set<int> ans;
    for (int a = 1; a * a * a <= N; a++) {
        for (int b = a; a * a * a + b * b * b <= N; b++) {
            ans.insert(a * a * a + b * b * b);
        }
    }

    bool first = true;
    for (int x : ans) {
        if (!first) cout << " ";
        cout << x;
        first = false;
    }
    cout << endl;
    return 0;
}
```

### 第 19 题（编程第4题）：密码破译

```cpp
#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

bool isPerfectSquare(int x) {
    int s = sqrt(x);
    return s * s == x;
}

int main() {
    int N;
    cin >> N;

    vector<pair<int, int>> ans;
    for (int a = 1; a * a <= N; a++) {
        for (int b = a; a * a + b * b <= N; b++) {
            int sum = a * a + b * b;
            if (isPerfectSquare(sum)) {
                ans.push_back({a, b});
            }
        }
    }

    if (ans.empty()) {
        cout << "No" << endl;
        return 0;
    }

    for (auto [a, b] : ans) {
        cout << a << " " << b << endl;
    }
    return 0;
}
```

### 第 20 题（编程第5题）：小华的零花钱

```cpp
#include <iostream>
using namespace std;

int main() {
    int N;
    cin >> N;

    long long total = 0;
    for (int day = 1; day <= N; day++) {
        total += day;
        if (total > 50) {
            total -= 10;
        }
    }

    cout << total << endl;
    return 0;
}
```
