```cpp
int main() {
    int n; cin >> n;
    vector<ll> d(n), v(n);
    for (int i = 0; i < n; i++) cin >> d[i] >> v[i];
    stack<pair<ll,ll>> st;
    ll h = 0, gold = 0;
    for (int i = 0; i < n; i++) {
        if (d[i] > h) { st.push({d[i], v[i]}); }
        else {
            gold += v[i]; h += d[i];
            while (!st.empty() && st.top().first <= h) {
                gold += st.top().second; h += st.top().first;
                st.pop();
            }
        }
    }
    while (!st.empty() && st.top().first <= h) {
        gold += st.top().second; h += st.top().first;
        st.pop();
    }
    cout << gold << endl;
```

19. cpp (100分)
参考代码：
```cpp
#include <bits/stdc++.h>
using namespace std;
const int MOD = 1e9 + 7;
const int MAXN = 1005;
int n, k, p[25];
int memo[MAXN];

int dfs(int x) {
    if (x == 0) return 1;          // ② 始：凑出 0，什么都不加就是一种方式
    if (x < 0) return 0;           // 边界：不能为负数
    if (memo[x] != -1) return memo[x]; // ③ 查：算过了直接返回

    long long ans = 0;
    for (int i = 0; i < k; i++) {
        if (p[i] <= x) {
            ans = (ans + dfs(x - p[i])) % MOD;
        }
    }
    memo[x] = ans;                 // ④ 录：记下来
    return memo[x];                // ⑤ 返
}

int main() {
    cin >> n >> k;
    for (int i = 0; i < k; i++) cin >> p[i];
    memset(memo, -1, sizeof(memo));
    cout << dfs(n) << endl;
    return 0;
}
```
20. cpp (100分)
参考代码：
```cpp
#include <bits/stdc++.h>
using namespace std;
const int MAXN = 105;
int n, m;
int a[MAXN][MAXN];
int memo[MAXN][MAXN];

int dfs(int i, int j) {
    if (i == n && j == m) return a[i][j];       // ② 始：到终点了
    if (memo[i][j] != -1) return memo[i][j];   // ③ 查：算过了直接返回

    int best = 0;
    if (i + 1 <= n)                     // 向下走
        best = max(best, dfs(i + 1, j));
    if (j + 1 <= m)                     // 向右走
        best = max(best, dfs(i, j + 1));
    if (i + 1 <= n && j + 1 <= m)       // 向右下（对角线）
        best = max(best, dfs(i + 1, j + 1));

    memo[i][j] = a[i][j] + best;       // ④ 录：当前格子 + 后续最佳
    return memo[i][j];                  // ⑤ 返
}

int main() {
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            cin >> a[i][j];
    memset(memo, -1, sizeof(memo));
    cout << dfs(1, 1) << endl;
    return 0;
}
```
