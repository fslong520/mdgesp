```cpp
#include <bits/stdc++.h>
using namespace std;
int a[105][105];
int main() {
    int n, m;
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            cin >> a[i][j];
    int maxRow = 1, maxLen = 0;
    for (int i = 1; i <= n; i++) {
        int cur = 0, best = 0;
        for (int j = 1; j <= m; j++) {
            if (a[i][j] == 0) cur++;
            else cur = 0;
            if (cur > best) best = cur;
        }
        if (best > maxLen) {
            maxLen = best;
            maxRow = i;
        }
    }
    cout << maxRow << " " << maxLen << endl;
    return 0;
}
```

参考代码:
```cpp
#include <bits/stdc++.h>
using namespace std;
struct Student {
    string name;
    int total, novel, science, id;
};
bool cmp(Student a, Student b) {
    if (a.total != b.total) return a.total > b.total;
    if (a.novel != b.novel) return a.novel > b.novel;
    if (a.science != b.science) return a.science > b.science;
    return a.id < b.id;
}
int main() {
    int n;
    cin >> n;
    Student s[55];
    for (int i = 0; i < n; i++) {
        cin >> s[i].name >> s[i].total >> s[i].novel >> s[i].science;
        s[i].id = i;
    }
    sort(s, s + n, cmp);
    for (int i = 0; i < n; i++)
        cout << s[i].name << endl;
    return 0;
}
```

```cpp
#include <bits/stdc++.h>
using namespace std;
const int MOD = 1e9 + 7;
const int MAXN = 100005;
int memo[MAXN][4];

int dfs(int x, int last) {
    if (x == 0) return 1;
    if (x < 0) return 0;
    if (memo[x][last] != -1) return memo[x][last];
    
    long long ans = 0;
    if (last == 1) {
        ans = ((long long)dfs(x - 2, 2) + dfs(x - 3, 3)) % MOD;
    } else {
        ans = ((long long)dfs(x - 1, 1) + dfs(x - 2, 2) + dfs(x - 3, 3)) % MOD;
    }
    memo[x][last] = ans;
    return ans;
}

int main() {
    int n;
    cin >> n;
    memset(memo, -1, sizeof(memo));
    cout << dfs(n, 0) << endl;
    return 0;
}
```
