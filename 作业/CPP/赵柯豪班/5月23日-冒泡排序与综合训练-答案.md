```cpp
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = 0; i < n; i++)
        for (int j = 1; j < n - i; j++)
            if (a[j] < a[j - 1])
                swap(a[j], a[j - 1]);
    bool first = true;
    for (int i = n - 1; i >= 0; i--) {
        if (a[i] % 2 == 1) {
            if (!first) cout << " ";
            cout << a[i];
            first = false;
        }
    }
    return 0;
}
```

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, a[505];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];
    for (int i = 0; i < n; i++)
        for (int j = 1; j < n - i; j++)
            if (a[j] < a[j - 1])
                swap(a[j], a[j - 1]);
    int mn = 100000;
    for (int i = 1; i < n; i++)
        if (a[i] - a[i - 1] < mn)
            mn = a[i] - a[i - 1];
    cout << mn;
    return 0;
}
```
