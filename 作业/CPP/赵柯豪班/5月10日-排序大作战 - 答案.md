## 编程题参考代码

### 16. 等差数列求和

```cpp
#include <iostream>
using namespace std;

int a, d, n;

int main() {
    cin >> a >> d >> n;
    int sum = n * a + n * (n - 1) * d / 2;
    cout << sum << endl;
    return 0;
}
```

### 18. 连续签到奖励

```cpp
#include <iostream>
using namespace std;

int n, sum = 0;

int main() {
    cin >> n;
    
    for (int i = 1; i <= n; i++) {
        sum += i;
    }
    
    cout << sum << endl;
    
    return 0;
}
```

### 19. 数组逆序存储

```cpp
#include <iostream>
using namespace std;

int n;
int a[105];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    for (int i = n - 1; i >= 0; i--) {
        cout << a[i] << (i == 0 ? "" : " ");
    }
    cout << endl;
    
    return 0;
}
```

### 20. 成绩排序

```cpp
#include <iostream>
#include <algorithm>
using namespace std;

int n;
int a[105];

int main() {
    cin >> n;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
    }
    
    // 冒泡排序
    for (int i = 0; i < n; i++) {
        for (int j = 1; j < n - i; j++) {
            if (a[j] < a[j-1]) {
                swap(a[j], a[j-1]);
            }
        }
    }
    
    for (int i = 0; i < n; i++) {
        cout << a[i] << (i == n-1 ? "" : " ");
    }
    cout << endl;
    
    return 0;
}
```
