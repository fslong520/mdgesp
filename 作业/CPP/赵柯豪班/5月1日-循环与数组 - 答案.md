### 17. 职业推荐

```cpp
#include <iostream>
using namespace std;

int main() {
    int str, agi, intl;
    cin >> str >> agi >> intl;
    
    // 找最大值
    int maxVal = str;
    if (agi > maxVal) maxVal = agi;
    if (intl > maxVal) maxVal = intl;
    
    // 统计有几个属性等于最大值
    int cnt = 0;
    if (str == maxVal) cnt++;
    if (agi == maxVal) cnt++;
    if (intl == maxVal) cnt++;
    
    if (cnt >= 2) {
        cout << "All-rounder" << endl;
    } else if (str == maxVal) {
        cout << "Warrior" << endl;
    } else if (agi == maxVal) {
        cout << "Archer" << endl;
    } else {
        cout << "Mage" << endl;
    }
    
    return 0;
}
```

### 19. 抽卡统计

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int cnt[5] = {0}; // cnt[1]=N卡, cnt[2]=R卡, cnt[3]=SR卡, cnt[4]=SSR卡
    
    for (int i = 0; i < n; i++) {
        int rarity;
        cin >> rarity;
        cnt[rarity]++;
    }
    
    for (int i = 1; i <= 4; i++) {
        cout << cnt[i] << (i == 4 ? "" : " ");
    }
    cout << endl;
    return 0;
}
```

### 20. 道具图鉴

```cpp
#include <iostream>
using namespace std;

int main() {
    int n;
    cin >> n;
    int vis[6] = {0}; // vis[i]=1 表示类型 i 出现过
    for (int i = 0; i < n; i++) {
        int t;
        cin >> t;
        vis[t] = 1;
    }
    int cnt = 0;
    for (int i = 1; i <= 5; i++) {
        if (vis[i]) cnt++;
    }
    cout << cnt << endl;
    return 0;
}
```
