using namespace std;
int main() {
    int n;
    cin >> n;
    int costA = n * 6;
    int groups = n / 6;
    int rem = n % 6;
    int costB = groups * 30 + rem * 6;
    if (costA < costB) cout << "A" << endl;
    else if (costA > costB) cout << "B" << endl;
    else cout << "Same" << endl;
    return 0;
}

    int maxv = 0, minv = 1000000;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        if (a[i] > maxv) maxv = a[i];
