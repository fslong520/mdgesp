bool _binarySearch(list<int> lst, int key) {
    if (lst.empty()) return false;
    auto left = lst.begin();
    auto right = prev(lst.end());
    while (distance(left, right) > 0) {
        auto mid = next(left, distance(left, right) / 2);

}

## 编程题 1【[小杨的幸运数](https://fslong.iok.la/problem/4459)】(25分)
小杨认为，所有大于等于某个值的完全平方数都是他的超级幸运数。
小杨还认为，所有超级幸运数的倍数都是他的幸运数。自然地，小杨的所有超级幸运数也都是幸运数。
对于一个非幸运数，小杨规定，可以将它一直进行某种操作，直到它变成一个幸运数。我们把这个过程叫做幸运化。
现在，小杨给出若干个数，请你首先判断它们是不是幸运数；接着，对于非幸运数，请你将它们幸运化。

**参考程序：**
```cpp
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <string>
#include <map>
#include <iostream>
#include <cmath>
using namespace std;
const int N = 1001 * 1001;
const double eps = 1e-8;
bool is_lucky[N + 5];
int next_lucky[N + 5];

void init() {
}

int main() {
    int a, T;
    scanf("%d%d", &a, &T);
    for (int i = 1; i <= N; i ++) {
        int t = int(sqrt(i) + eps);
        if (i >= a && t * t == i)
            is_lucky[i] = 1;
        if (! is_lucky[i])
            continue ;
        
        for (int j = i + i; j <= N; j += i)
            is_lucky[j] = 1;
    }
    for(int i = N; i; i --)
        next_lucky[i] = is_lucky[i] ? i : next_lucky[i + 1];
    
    while(T --) {
        int x;
        scanf("%d", &x);
        if (is_lucky[x])
            cout << "lucky" << endl;
        else
            cout << next_lucky[x] << endl;
    }
    return 0;
}
```

## 编程题 2【[烹饪问题](https://fslong.iok.la/problem/4460)】(25分)
有n种食材，编号从1至n，其中第i种食材的美味度为a_i。
不同食材之间的组合可能产生奇妙的化学反应。具体来说，如果两种食材的美味度分别为x和y，那么它们的契合度为x & y（按位与运算）。
现在，请你找到契合度最高的两种食材，并输出它们的契合度。

**参考程序：**
```cpp
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstdlib>
#include<cstring>
using namespace std;
const int MAX_N = int(1e6) + 100;

int a[MAX_N];

int sort(int l,int r,int k) {
    while(l <= r) {
        while ((l <= r) && (a[l] >> k & 1)) l++;
        while ((l <= r) && (!(a[r] >> k & 1))) r--;
        if (l <= r) swap(a[l++], a[r--]);
    }
    return r;
}

int main() {
    int n, j, ans=0;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    for (int i = 31; i >= 0; i--)
        if ((j = sort(1, n, i)) >= 2) {
            ans = ans | 1 << i;
            n = j;
        }
    printf("%d\n", ans);
    return 0;
}
```
